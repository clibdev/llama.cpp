docker run -it --rm --entrypoint=bash --name=cuda-container nvidia/cuda:13.0.3-cudnn-devel-ubuntu24.04

docker cp cuda-container:/usr/local/cuda/ ./cuda-runtime

rm -rf build

CUDA_PATH=$PWD/cuda-runtime cmake -B build -G Ninja -DCMAKE_BUILD_TYPE=Release -DCMAKE_INSTALL_PREFIX=build/out \
    -DBUILD_SHARED_LIBS=OFF \
    -DGGML_STATIC=ON \
    -DCMAKE_EXE_LINKER_FLAGS='-static-libstdc++ -static-libgcc' \
    -DCMAKE_INTERPROCEDURAL_OPTIMIZATION=ON \
    -DGGML_NATIVE=OFF \
    -DLLAMA_OPENSSL=OFF \
    -DLLAMA_BUILD_EXAMPLES=OFF \
    -DLLAMA_BUILD_TESTS=OFF \
    -DLLAMA_BUILD_UI=OFF \
    -DGGML_CUDA=ON

cmake --build build -j$(nproc)
cmake --install build --strip

./build/out/bin/llama-server -m Qwen3.6-27B-UD-Q4_K_XL.gguf -c 262144 -ngl 99 -np 1 -fa on --cache-type-k q4_0 --cache-type-v q4_0
